		T0.AvgPrice As 'Costo', 
		t0.Onhand As 'Saldo'
		 ,T2.WhsName
		, t3.ItmsGrpNam
from OITW T0 
INNER JOIN OITM T1 ON t0.ItemCode = t1.ItemCode
Inner join OWHS t2 on T0.WhsCode = t2.WhsCode
